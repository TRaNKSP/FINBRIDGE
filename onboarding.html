Step 2 — Install the SDK

# Python pip install finbridge-sdk # Node.js npm install @finbridge/sdk # Java <dependency> <groupId>io.finbridge</groupId> <artifactId>finbridge-java</artifactId> <version>1.4.2</version> </dependency>
# Python — initialize client from finbridge import FinBridgeClient client = FinBridgeClient( api_key="fb2030_sk_YOUR_KEY_HERE", environment="sandbox" # or "production" ) # Validate connection status = client.auth.validate() print(status.org_name) # → "Acme Community Bank"

Step 3 — Run your first sandbox call

# Test: Apply for a Gig Micro-Credit line result = client.micro_credit.apply( worker_id="WKR-89234", platform="uber", requested_amount=500, purpose="vehicle_maintenance" ) print(result.application_id) # → MCA-20261104-00918 print(result.status) # → "approved" print(result.approved_amount) # → 500 print(result.ai_score) # → 724
# Test: Subscribe to webhooks webhook = client.webhooks.subscribe( url="https://your-server.com/finbridge-events", events=[ "loan.approved", "loan.repayment_made", "fuel_advance.triggered", "bridge.distribution_issued" ] ) print(webhook.signing_secret) # Store this to verify webhook signatures
# Verify webhook signatures (Python + Flask) import hmac, hashlib def verify_webhook(payload_bytes, signature_header, secret): expected = hmac.new( secret.encode(), payload_bytes, hashlib.sha256 ).hexdigest() return hmac.compare_digest("sha256=" + expected, signature_header) # In your Flask handler: @app.route("/finbridge-events", methods=["POST"]) def handle_webhook(): payload = request.get_data() sig = request.headers.get("X-FinBridge-Signature") if not verify_webhook(payload, sig, WEBHOOK_SECRET): return "Unauthorized", 401 event = request.get_json() if event["event"] == "loan.approved": handle_loan_approved(event["data"]) return "OK", 200
